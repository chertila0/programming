import matplotlib.pyplot as plt

#### –ó–∞–¥–∞—á–∞ 2

–î–ª—è –∑–∞–¥–∞–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ ùëì(ùë•) –∞–Ω–∞–ª–∏—Ç–∏—á–µ—Å–∫–∏ –Ω–∞–π—Ç–∏ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –∫–∞—Å–∞—Ç–µ–ª—å–Ω–æ–π –ø—Ä—è–º–æ–π –∏ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π –ø—Ä—è–º–æ–π –≤ —É–∫–∞–∑–∞–Ω–Ω–æ–π —Ç–æ—á–∫–µ ùë•0. –ò—Å–ø–æ–ª—å–∑—É—è –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–µ –ø–∞–∫–µ—Ç—ã Python –ø–æ—Å—Ç—Ä–æ–∏—Ç—å –Ω–∞ –æ–¥–Ω–æ–º —Ä–∏—Å—É–Ω–∫–µ –≥—Ä–∞—Ñ–∏–∫ —Ñ—É–Ω–∫—Ü–∏–∏ ùëì(ùë•), –∫–∞—Å–∞—Ç–µ–ª—å–Ω—É—é –ø—Ä—è–º—É—é, –Ω–æ—Ä–º–∞–ª—å–Ω—É—é –ø—Ä—è–º—É—é –∏ –æ—Ç–º–µ—Ç–∏—Ç—å —Ç–æ—á–∫—É –∫–∞—Å–∞–Ω–∏—è.

$f(x) = xsinx$

$x_0 = \pi$

#### –ê–Ω–∞–ª–∏—Ç–∏—á–µ—Å–∫–æ–µ —Ä–µ—à–µ–Ω–∏–µ
–ù–∞–π–¥–µ–º –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—É—é –¥–∞–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏:

$(xsinx)' = x'sinx + x(sinx)'= sinx+xcosx$

–ù–∞–π–¥–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–æ–π –≤ —Ç–æ—á–∫–µ $x_0=\pi$:

$f'(x_0)=f'(\pi)= sin\pi+\pi cos\pi = -\pi$

–ù–∞–π–¥–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –≤ —Ç–æ—á–∫–µ $x_0$:

$f(x_0)=\pi sin\pi = 0$

–°–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ, —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –∫–∞—Å–∞—Ç–µ–ª—å–Ω–æ–π –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –≤–∏–¥:

$y = f(x_0)+f'(x_0)(x-x_0)= -\pi(x-\pi)$

–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π –ø—Ä—è–º–æ–π –±—É–¥–µ—Ç –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω—ã–º –∫–∞—Å–∞—Ç–µ–ª—å–Ω–æ–π –ø—Ä—è–º–æ–π –∏ –ø—Ä–æ—Ö–æ–¥—è—â–∏–º —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É –∫–∞—Å–∞–Ω–∏—è.

–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–π –ø—Ä—è–º–æ–π:

$y=f(x_0)-\frac{1}{f'(x_0)}(x-x_0) = \frac{1}{\pi}(x-\pi)$

#### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ

```python
import matplotlib.pyplot as plt 
import numpy as np


def f(x):
    return x * np.sin(x)

def kasat(x):
    return -np.pi*(x-np.pi)

def normal(x):
    return 1/np.pi*(x-np.pi)

x = np.linspace(-7,7,70)
y = f(x)

plt.plot(x,y, label = 'f(x)')
plt.plot(x,kasat(x), label = '–ö–∞—Å–∞—Ç–µ–ª—å–Ω–∞—è')
plt.plot(x,normal(x), label = '–ù–æ—Ä–º–∞–ª—å–Ω–∞—è –ø—Ä—è–º–∞—è')
plt.plot(np.pi, f(np.pi), 'ro', label = '–¢–æ—á–∫–∞ –∫–∞—Å–∞–Ω–∏—è')
plt.gca().set_aspect('equal')
plt.legend()
plt.grid()
plt.show()

```
