### –ó–∞–¥–∞–Ω–∏–µ 1
–î–∞–Ω—ã –¥–≤–µ —Ç–æ—á–∫–∏ $A = (1,2)$ –∏ $B = (2,5)$. –ù–∞–ø–∏—Å–∞—Ç—å –ø—Ä–æ–≥—Ä–∞–º–º—É –Ω–∞ 
—è–∑—ã–∫–µ Python —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –±–∏–±–ª–∏–æ—Ç–µ–∫ matplotlib –∏ numpy, –∫–æ—Ç–æ—Ä–∞—è —Å—Ç—Ä–æ–∏—Ç —Ä–∏—Å—É–Ω–æ–∫, –∏–∑–æ–±—Ä–∞—â–∞—é—â–∏–π –ø—Ä—è–º—É—é, –ø—Ä–æ—Ö–æ–¥—è—â—É—é —á–µ—Ä–µ–∑ —ç—Ç–∏ —Ç–æ—á–∫–∏, –∞ —Ç–∞–∫–∂–µ –µ–¥–∏–Ω–∏—á–Ω—ã–π –≤–µ–∫—Ç–æ—Ä $n$ –Ω–æ—Ä–º–∞–ª–∏ —ç—Ç–æ–π –ø—Ä—è–º–æ–π. –ù–∞ —Ä–∏—Å—É–Ω–∫–µ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –ø–æ–¥–ø–∏—Å–∞–Ω—ã –æ—Å–∏ $Ox$ –∏ $Oy$, —Ç–æ—á–∫–∏ $A$ –∏ $B$ –∏ –≤–µ–∫—Ç–æ—Ä $n$. –ü–æ—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π —Ä–∏—Å—É–Ω–æ–∫ –¥–æ–ª–∂–µ–Ω –ø—Ä–∏–±–ª–∏–∑–∏—Ç–µ–ª—å–Ω–æ –∏–º–µ—Ç—å –≤–∏–¥, –ø–æ–∫–∞–∑–∞–Ω–Ω—ã–π –Ω–∞ —Ä–∏—Å. 1. –¢–∞–∫–∂–µ –ø—Ä–æ–≥—Ä–∞–º–º–∞ –¥–æ–ª–∂–Ω–∞ –≤—ã–≤–æ–¥–∏—Ç—å —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–æ–π –ø—Ä—è–º–æ–π –≤ –≤–∏–¥–µ $y = kx + b$.

#### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ
```python
import matplotlib.pyplot as plt 
import numpy as np
if __name__ == "__main__":
    A = np.array([1, 2])
    B = np.array([2, 5])
def plot_pic():
    plt.plot([A[0], B[0]], [A[1], B[1]], color='blue')
    plt.scatter(A[0], A[1], color='red', alpha=1)
    plt.scatter(B[0], B[1], color='red', alpha=1)
    normal = np.array([B[1] - A[1], A[0] - B[0]])
    mid_point = (A + B) / 2
    plt.arrow(mid_point[0], mid_point[1], normal[0], normal[1], color='red',
head_width=0.15, head_length=0.3, fc='red', ec='red')
    plt.text(A[0] + 0.4, A[1], 'A', fontsize=12, ha='right')
    plt.text(B[0] - 0.2, B[1], 'B', fontsize=12, ha='right')
    plt.text(mid_point[0] + normal[0]/2, mid_point[1] + normal[1]/2 + 0.2, 'n',
fontsize=12, ha='right')
    plt.xlabel("x", fontsize="xx-large",
 fontstyle="italic", family="serif")
    plt.ylabel("y", fontsize="xx-large",
 fontstyle="italic", family="serif")
    plt.show()
plot_pic()
# –ø–æ–ª—É—á–µ–Ω–∏–µ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –ø—Ä—è–º–æ–π
k = (B[1] - A[1]) / (B[0] - A[0])
b = A[1] - k * A[0]
print(f"–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä—è–º–æ–π: y = {k}x + ({b})")

```

### –ó–∞–¥–∞–Ω–∏–µ 2

–î–∞–Ω–∞ —Ç–æ—á–∫–∞ $ùê¥ = (‚àí4, ‚àí3)$ –∏ –≤–µ–∫—Ç–æ—Ä $u = (7, 1)$. –ù–∞–ø–∏—Å–∞—Ç—å –ø—Ä–æ–≥—Ä–∞–º–º—É –Ω–∞ —è–∑—ã–∫–µ Python —Å–∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –±–∏–±–ª–∏–æ—Ç–µ–∫ matplotlib –∏ numpy, –∫–æ—Ç–æ—Ä–∞—è —Å—Ç—Ä–æ–∏—Ç —Ä–∏—Å—É–Ω–æ–∫,–∏–∑–æ–±—Ä–∞—â–∞—é—â–∏–π –ø—Ä—è–º—É—é, –ø—Ä–æ—Ö–æ–¥—è—â—É—é —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É $ùê¥$ –≤ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ –≤–µ–∫—Ç–æ—Ä–∞ $u$, –∞—Ç–∞–∫–∂–µ –µ–¥–∏–Ω–∏—á–Ω—ã–π –≤–µ–∫—Ç–æ—Ä $n$ –Ω–æ—Ä–º–∞–ª–∏ —ç—Ç–æ–π –ø—Ä—è–º–æ–π. –ù–∞ —Ä–∏—Å—É–Ω–∫–µ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å–ø–æ–¥–ø–∏—Å–∞–Ω—ã –æ—Å–∏ $ùëÇùë•$ –∏ $ùëÇùë¶$, —Ç–æ—á–∫–∞ $ùê¥$ –∏ –≤–µ–∫—Ç–æ—Ä—ã $u$ –∏ $n$. –¢–∞–∫–∂–µ –ø—Ä–æ–≥—Ä–∞–º–º–∞ –¥–æ–ª–∂–Ω–∞ –≤—ã–≤–æ–¥–∏—Ç—å —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–æ–π –ø—Ä—è–º–æ–π –≤ –≤–∏–¥–µ $ùë¶ = ùëòùë• + ùëè$.

#### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ

```python
import numpy as np
import matplotlib.pyplot as plt
# –ó–∞–¥–∞–µ–º —Ç–æ—á–∫—É A –∏ –≤–µ–∫—Ç–æ—Ä u
A = np.array([-4, -3])
u = np.array([7, 1])
# –í—ã—á–∏—Å–ª—è–µ–º –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –∫–æ–Ω–µ—á–Ω–æ–π —Ç–æ—á–∫–∏ –ø—Ä—è–º–æ–π
B = A + u
# –í—ã—á–∏—Å–ª—è–µ–º –Ω–æ—Ä–º–∞–ª—å–Ω—ã–π –≤–µ–∫—Ç–æ—Ä –∫ –ø—Ä—è–º–æ–π
n = np.array([-u[1], u[0]])
# –í—ã—á–∏—Å–ª—è–µ–º –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã —É—Ä–∞–≤–Ω–µ–Ω–∏—è –ø—Ä—è–º–æ–π y = kx + b
k = u[1] / u[0]
b = A[1] - k * A[0]
# –°–æ–∑–¥–∞–µ–º —Ñ–∏–≥—É—Ä—É –∏ –æ—Å–∏
fig, ax = plt.subplots()
# –†–∏—Å—É–µ–º –ø—Ä—è–º—É—é, –ø—Ä–æ—Ö–æ–¥—è—â—É—é —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É A –≤ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ –≤–µ–∫—Ç–æ—Ä–∞ u
ax.plot([A[0], B[0]], [A[1], B[1]], 'r', label='–ü—Ä—è–º–∞—è')
# –†–∏—Å—É–µ–º –Ω–æ—Ä–º–∞–ª—å–Ω—ã–π –≤–µ–∫—Ç–æ—Ä
ax.arrow(A[0], A[1], n[0], n[1], head_width=0.2, head_length=0.2, fc='b', ec='b',
label='–ù–æ—Ä–º–∞–ª—å')
# –ü–æ–¥–ø–∏—Å—ã–≤–∞–µ–º –æ—Å–∏
ax.set_xlabel('x')
ax.set_ylabel('y')
# –ü–æ–¥–ø–∏—Å—ã–≤–∞–µ–º —Ç–æ—á–∫—É A –∏ –≤–µ–∫—Ç–æ—Ä—ã u –∏ n
ax.annotate('A', A, textcoords="offset points", xytext=(-6,-14), ha='center')
ax.annotate('u', B, textcoords="offset points",xytext=(8, 4), ha='center')
ax.annotate('n', A + n, textcoords="offset points", xytext=(-10,10), ha='center')
# –í—ã–≤–æ–¥–∏–º —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä—è–º–æ–π –≤ —Ç–µ—Ä–º–∏–Ω–∞–ª–µ
print(f'–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä—è–º–æ–π: y = {round(k, 2)}x + {round(b,2)}')
# –í—ã–≤–æ–¥–∏–º –≥—Ä–∞—Ñ–∏–∫
plt.legend()
plt.grid(True)
plt.show()

```

### –ó–∞–¥–∞–Ω–∏–µ 3
–î–∞–Ω–∞ —Ç–æ—á–∫–∞ $ùê¥ = (‚àí2, 7)$ –∏ –≤–µ–∫—Ç–æ—Ä $u = (-1, -1)$. –ù–∞–ø–∏—Å–∞—Ç—å –ø—Ä–æ–≥—Ä–∞–º–º—É –Ω–∞ —è–∑—ã–∫–µ Python —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –±–∏–±–ª–∏–æ—Ç–µ–∫ matplotlib –∏ numpy, –∫–æ—Ç–æ—Ä–∞—è —Å—Ç—Ä–æ–∏—Ç —Ä–∏—Å—É–Ω–æ–∫, –∏–∑–æ–±—Ä–∞—â–∞—é—â–∏–π –ø—Ä—è–º—É—é, –ø—Ä–æ—Ö–æ–¥—è—â—É—é —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É $ùê¥$ –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω–æ –≤–µ–∫—Ç–æ—Ä—É $u$. –ù–∞ —Ä–∏—Å—É–Ω–∫–µ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –ø–æ–¥–ø–∏—Å–∞–Ω—ã –æ—Å–∏ $ùëÇùë•$ –∏ $ùëÇùë¶$, —Ç–æ—á–∫–∞ $ùê¥$ –∏ –≤–µ–∫—Ç–æ—Ä $u$. –¢–∞–∫–∂–µ –ø—Ä–æ–≥—Ä–∞–º–º–∞ –¥–æ–ª–∂–Ω–∞ –≤—ã–≤–æ–¥–∏—Ç—å —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–æ–π –ø—Ä—è–º–æ–π –≤ –≤–∏–¥–µ $ùë¶ = ùëòùë• + ùëè$.

#### –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ

```python
import numpy as np
import matplotlib.pyplot as plt
# –ó–∞–¥–∞–µ–º —Ç–æ—á–∫—É A –∏ –≤–µ–∫—Ç–æ—Ä u
A = np.array([-2, 7])
u = np.array([-1, -1])
# –ù–∞—Ö–æ–¥–∏–º –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –ø—Ä—è–º–æ–π
k = -u[0] / u[1]
b = A[1] - k * A[0]
# –°–æ–∑–¥–∞–µ–º –º–∞—Å—Å–∏–≤ –∑–Ω–∞—á–µ–Ω–∏–π x
x = np.linspace(-10, 10, 100)
# –í—ã—á–∏—Å–ª—è–µ–º —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è y
y = k * x + b
# –°—Ç—Ä–æ–∏–º –≥—Ä–∞—Ñ–∏–∫
plt.plot(x, y, label='y = {}x + {}'.format(k, b))
plt.scatter(A[0], A[1], color='red', label='–¢–æ—á–∫–∞ A')
plt.arrow(A[0], A[1], u[0], u[1], color='blue', width=0.1,
head_width=0.5, length_includes_head=True, label='–í–µ–∫—Ç–æ—Ä u')
plt.xlabel('x')
plt.ylabel('y')
plt.title('–ü—Ä—è–º–∞—è, –ø—Ä–æ—Ö–æ–¥—è—â–∞—è —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É A –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω–æ –≤–µ–∫—Ç–æ—Ä—É u')
plt.legend()
plt.grid(True)
plt.show()
# –í—ã–≤–æ–¥–∏–º —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä—è–º–æ–π
print('–£—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä—è–º–æ–π: y = {}x + {}'.format(k, b))
```
